Git дҪҝз”Ёе°ҸжҠҖе·§ вҖӢ

еҲӣе»әдёҖдёӘе№ІеҮҖзҡ„еҲҶж”ҜпјҲж— еҺҶеҸІи®°еҪ•зҡ„еҲҶж”Ҝпјү
git checkout --orphan branchNane

# еҹәдәҺж Үзӯҫ v1.0 еҲӣе»әе№¶еҲҮжҚўеҲ°ж–°еҲҶж”Ҝ
git checkout -b branchNane tagName 
# жҲ–иҖ… git switch -c branchNane tagName

# gitдјҡйҖҖеҲ°жҢҮе®ҡзҡ„commitе№¶дё”дҝқз•ҷд№ӢеүҚдҝ®ж”№зҡ„ж–Үд»¶
git reset --soft <commit-hash>
git reset --soft HEAD^  ж’Өй”ҖдёҠдёҖж¬Ўзҡ„commit

# з§»еҠЁ HEAD жҢҮй’ҲпјҢе№¶жё…з©әжҡӮеӯҳеҢәпјҢдҪҶдҝқз•ҷе·ҘдҪңеҢәзҡ„еҶ…е®№гҖӮ
git reset --mixed <commit>
# зӯүд»·дәҺ
git reset <commit>

# з§»еҠЁ HEAD жҢҮй’ҲпјҢеҗҢж—¶жё…з©әжҡӮеӯҳеҢәпјҢе№¶е°Ҷе·ҘдҪңеҢәжҒўеӨҚеҲ°жҢҮе®ҡжҸҗдәӨзҡ„зҠ¶жҖҒгҖӮ
git reset --hard <commit>

git rm --cached гҖҗж–Үд»¶и·Ҝеҫ„гҖ‘    // е°Ҷж–Үд»¶д»Һзј“еӯҳеҢәдёӯеҲ йҷӨпјҢе№¶дҝқз•ҷж–Үд»¶

# з»ҷеҲҶж”ҜйҮҚж–°е‘ҪеҗҚ
git branch -m old-name new-name

# з»ҲжӯўеҸҳеҹә
git rebase --abort

# з»Ҳжӯўmerge
git merge --abort

# жҹҘзңӢе…ЁеұҖй…ҚзҪ®
git config --list

# дҝ®ж”№е…ЁеұҖй…ҚзҪ® -- дҝ®ж”№еҪ“еүҚз”ЁжҲ·еҗҚ
git config --global user.name "Your New Name"

# дҝ®ж”№е…ЁеұҖй…ҚзҪ® -- дҝ®ж”№еҪ“еүҚйӮ®з®ұ
git config --global user.email "your.email@example.com"

# е°ҶжҸҗдәӨеӨҚеҲ¶еҲ°дёҚеҗҢзҡ„еҲҶж”ҜпјҢ commitId еҸҜд»ҘйҖҡиҝҮ git log иҺ·еҸ–
git cherry-pick commitId

# commitId еҸҜд»ҘйҖҡиҝҮ git log иҺ·еҸ–пјҢ

# ж”ҫејғпјҲж’Өй”ҖпјүжңӘ git add зҡ„ж–Үд»¶дҝ®ж”№
git restore .пјҡжҒўеӨҚжүҖжңүжңӘжҡӮеӯҳзҡ„дҝ®ж”№пјҢжҒўеӨҚеҗҺж–Үд»¶дјҡеӣһеҲ°дёҠдёҖж¬ЎжҸҗдәӨзҡ„зҠ¶жҖҒгҖӮ
git restore <file_name>пјҡеҸӘжҒўеӨҚжҢҮе®ҡж–Үд»¶зҡ„жӣҙж”№гҖӮ

# жҳҺжҳҺе·Із»Ҹжӣҙж–°дәҶ.gitignoreж–Үд»¶пјҢдҪҶжҳҜиҝҳжҳҜеҸҜд»ҘзңӢеҲ°ж–Үд»¶зҡ„и·ҹиёӘи®°еҪ•пјҢжӯӨж—¶еҸҜиғҪжҳҜgitignoreжңүзј“еӯҳдәҶпјҢйңҖиҰҒжё…зҗҶзј“еӯҳд№ӢеҗҺеҶҚејҖе§Ӣж·»еҠ 
git rm -r cached .
git add .
git commit -m "жӣҙж–° .gitignore"

# дҪҝз”Ёgit еҗҲе№¶еӨҡдёӘcommitId
# йҰ–е…ҲйңҖиҰҒеӣһеҲ°иҰҒеӣһеҲ°зҡ„commitд№ӢеүҚпјҢжҜ”е–»иҰҒеҗҲе№¶3дёӘпјҢе°ұиҰҒеӣһеҲ°еҫҖеүҚз¬¬ 4 дёӘcommitId
###  дёҖе®ҡиҰҒиҝһз»ӯжҸҗдәӨзҡ„ commit жүҚиғҪиҝҷд№ҲеҒҡ
git rebase -i commitId

#еҪ“дҪ еҠ дёҠ -iпјҢе°ұжҳҜиҝӣе…Ҙ дәӨдә’ејҸ rebase жЁЎејҸгҖӮ
#еңЁиҝҷдёӘжЁЎејҸдёӢпјҢGit дјҡжү“ејҖдёҖдёӘзј–иҫ‘еҷЁпјҢеҲ—еҮәдёҖж®өжҸҗдәӨеҺҶеҸІпјҢдҪ еҸҜд»ҘеҜ№иҝҷдәӣ commit еҒҡеҗ„з§Қж“ҚдҪңпјҡ

#еёёи§Ғе‘Ҫд»Өпјҡ
#	вҖў	pickпјҡдҝқз•ҷиҝҷдёӘ commit
#	вҖў	rewordпјҡдҝ®ж”№ commit message
#	вҖў	editпјҡиҝӣе…ҘиҝҷдёӘ commitпјҢе…Ғи®ёдҪ ж”№д»Јз ҒеҶҚз»§з»ӯ
#	вҖў	squashпјҡжҠҠеҪ“еүҚ commit еҗҲе№¶еҲ°дёҠдёҖдёӘ commitпјҲеёёз”ЁжқҘвҖңеҗҲе№¶еӨҡдёӘ commitвҖқпјү
#	вҖў	fixupпјҡзұ»дјј squashпјҢдҪҶдјҡдёўејғеҪ“еүҚ commit зҡ„ message
#	вҖў	dropпјҡдёўејғиҝҷдёӘ commit

еҒҮи®ҫжңү3ж¬ЎжҸҗдәӨ
a1b2c3  дҝ®еӨҚ bug
d4e5f6  и°ғж•ҙж ·ејҸ
g7h8i9  жӣҙж–°ж–ҮжЎЈ

git rebase -i HEAD~3 зҪ‘дёҠеӣһеҲ°з¬¬ 4 ж¬Ў

жӯӨж—¶Git дјҡжү“ејҖзј–иҫ‘еҷЁпјҡ
pick a1b2c3 дҝ®еӨҚ bug
pick d4e5f6 и°ғж•ҙж ·ејҸ
pick g7h8i9 жӣҙж–°ж–ҮжЎЈ

дҝ®ж”№жҲҗ 
pick a1b2c3 дҝ®еӨҚ bug
squash d4e5f6 и°ғж•ҙж ·ејҸ
squash g7h8i9 жӣҙж–°ж–ҮжЎЈ

дҝқеӯҳеҗҺ вҶ’ иҝҷ 3 дёӘ commit дјҡеҗҲе№¶жҲҗ 1 дёӘж–°зҡ„ commitпјҢеҸҜд»ҘеңЁжӯӨж—¶дҝ®ж”№ commit еҶ…е®№гҖӮ

git rebase -i = дәӨдә’ејҸж”№еҶҷжҸҗдәӨеҺҶеҸІпјҢи®©дҪ еҸҜд»Ҙ еҗҲе№¶гҖҒдҝ®ж”№гҖҒеҲ йҷӨгҖҒйҮҚжҺ’ commitпјҢеёёз”ЁдәҺ ж•ҙзҗҶжҸҗдәӨи®°еҪ•гҖӮ

жҺҘзқҖжҲ‘д»¬еҸҜд»ҘдҪҝз”Ё дҪҝз”Ё git commit --amend жқҘдҝ®ж”№ commit еҶ…е®№пјҢе°ҶдёҠйқў3ж¬Ўзҡ„commitеҶ…е®№еҗҲе№¶жҲҗ1дёӘпјҢ
д№ҹеҸҜд»ҘдҪҝз”Ё git commit --amend --no-edit жқҘиҰҶзӣ–дёҠдёҖж¬ЎжҸҗдәӨпјҡ
д№ӢеҗҺе°ұжҳҜжӯЈеёёзҡ„ git pushж“ҚдҪңдәҶ

// е°ҶеҲҶж”Ҝ rebase merge еҲ°dev еҗҢдәӢдёҚдә§з”ҹmergeи®°еҪ•пјҢеҸӘжңүfeatureдёӯзҡ„жҜҸж¬ЎcommitжҸҗдәӨи®°еҪ•
git checkout dev && git rebase feature

// жғіиҝңзЁӢеҗҢжӯҘ dev ж—¶д№ҹдёҚдә§з”ҹпјҢ жӯӨж—¶жң¬ең°зҡ„еҲҶж”ҜжҳҜ feature 
git pull --rebase origin dev

// жғіеҸӘдҝқз•ҷдёҖдёӘжҖ»жҸҗдәӨпјҲдёҚдҝқз•ҷдёӯй—ҙ commitпјү
git merge --squash feature
